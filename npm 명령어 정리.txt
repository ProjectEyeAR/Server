npm init //npm 초기설정
npm init --yes //초기설정, 입력없이
npm list --depth=0 //설치된 패키지 보기
npm install //깃헙에서 처음 불러왔다면
npm view mongoose dependencies // 패키지 의존성 보기
npm view mongoose versions //패키지 모든 버전 보기
npm install mongoose@2.4.2 //패키지 버전 바꾸기(다운그레이드, 업그레이드)
npm outdated //패키지 업그레이드 확인
npm update //모든 패키지 wanted버전으로 업그레이드
npm install -g npm-check-updates
npm-check-updates
npm install //모든 패키지 최신버전으로 업그레이드
npm install jshint --save-dev //개발할때만 사용하는 패키지 설정
npm un mongoose //패키지 삭제

패키지 출판하기
npm adduser //npm에 회원가입
npm login //npm에 로그인
npm publish //패키지 출판
npm version major
npm version minor
npm version patch
